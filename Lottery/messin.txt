<script>
//function that grabs the numbers using getElementByID
function grabNumbers() {

input1 = document.getElementByID('input1').value;
input2 = document.getElementByID('input2').value;
input3 = document.getElementByID('input3').value;
input4 = document.getElementByID('input4').value;
input5 = document.getElementByID('input5').value;
}
//event listener that grabs numbers when they are changed by the user 
//(not when they are clicked)
document.getElementById('input1').addEventListener('onchange',getNumbers);
document.getElementById('input2').addEventListener('onchange',getNumbers);
document.getElementById('input3').addEventListener('onchange',getNumbers);
document.getElementById('input4').addEventListener('onchange',getNumbers);
document.getElementById('input5').addEventListener('onchange',getNumbers);


</script>
